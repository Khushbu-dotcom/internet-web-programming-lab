<?php

// Function to validate input
function validateInput($data) {
    return trim(htmlspecialchars($data));
}

// Function to calculate total expense for selected date
function calculateTotal($expenses, $selectedDate) {
    $total = 0;

    foreach ($expenses as $expense) {
        $parts = explode("|", $expense);

        if (trim($parts[0]) == $selectedDate) {
            $total += (float) trim($parts[2]);
        }
    }

    return $total;
}

// Save data to file
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $date = validateInput($_POST['date']);
    $category = validateInput($_POST['category']);
    $amount = validateInput($_POST['amount']);

    if (!empty($date) && !empty($category) && !empty($amount)) {
        $entry = "$date|$category|$amount\n";
        file_put_contents("expenses.txt", $entry, FILE_APPEND);
    }
}

// Read file into array
$expenses = file_exists("expenses.txt") ? file("expenses.txt") : [];

// Today's date
$today = date("Y-m-d");

?>

<!DOCTYPE html>
<html>
<head>
    <title>Daily Expense Tracker</title>
</head>
<body>

<h2>Expense Entry Form</h2>

<form method="POST">
    Date: <input type="date" name="date" required><br><br>

    Category:
    <input type="text" name="category" required><br><br>

    Amount:
    <input type="number" name="amount" required><br><br>

    <button type="submit">Add Expense</button>
</form>

<hr>

<h2>Today's Expenses (<?php echo $today; ?>)</h2>

<?php
foreach ($expenses as $expense) {
    $parts = explode("|", $expense);

    if (trim($parts[0]) == $today) {
        echo "Date: " . $parts[0] . " | Category: " . $parts[1] . " | Amount: ₹" . $parts[2] . "<br>";
    }
}
?>

<hr>

<h2>Total Expense for Today</h2>

<?php
echo "₹" . calculateTotal($expenses, $today);
?>

<hr>

<h2>All Stored Expenses</h2>

<?php
foreach ($expenses as $expense) {
    $parts = explode("|", $expense);

    echo "Date: " . $parts[0] . " | Category: " . $parts[1] . " | Amount: ₹" . $parts[2] . "<br>";
}
?>
